1.1 Região Negativa em uma Imagem

O primeiro exercício consiste em trabalhar com escala de cinza e gerar o negativo de uma dada região especificada pelo usuário através da coordenada de dois pontos P1 e P2 que estão localizados dentro dos limites da imagem.

Foi realizada a operação de subtração entre o nível máximo de tom de cinza que é de 255 menos o valor do pixel que será trabalhado no momento pelo código.

Código

Codigo 1 - regions.cpp
#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

int main(int, char**){
  Mat image;
  Vec3b val;
  int P1, P2;

  image= imread("biel.png",CV_LOAD_IMAGE_GRAYSCALE);

  if(!image.data)
    cout << "Não abriu a imagem" << endl;

  cout << "Escolha um valor para P1 entre 0 e " << image.rows << " e um valor para P2 entre 0 e " << image.cols << endl;
  cout<< "P1 : " << endl;
  cin >> P1;

  cout<< "P2 : " << endl;
  cin >> P2;

  for(int i=P1;i<P2;i++){
    for(int j=P1;j<P2;j++){
	image.at<uchar>(i,j)=255-image.at<uchar>(i,j);
    }
  }

  imshow("Região negativa", image);
  imwrite("imagem_negativa.png", image);
  waitKey();
  return 0;
}

1.2. Troca de Quadrantes

O segundo exercício consiste em trocar os quadrantes de uma imagem, considerando que essa imagem será dividida em quatro quadrantes e eles serão trocados em diagonal.

Para isso, utiliza-se das funções Mat() e Rect() fornecidas pelo OpenCV para definir as regiões e a função copyTo() para fazer a troca entre as regiões definidas no primeiro momento do código.

Código

Código 2 - trocaregioes.cpp
#include <iostream>
#include <opencv2/opencv.hpp>


using namespace cv;
using namespace std;

int main(int, char**){
  Mat image;
  unsigned char aux;


  image= imread("biel.png",CV_LOAD_IMAGE_GRAYSCALE);

   int linhas =image.size().width;
   int colunas=image.size().height;

  if(!image.data)
    cout << "nao abriu imagem" << endl;

  namedWindow("Troca Regiões",WINDOW_AUTOSIZE);

  for(int i=0;i<colunas/2;i++){
    for(int j=0;j<linhas/2;j++){
     	aux = image.at<uchar>(i,j);
	image.at<uchar>(i,j) = image.at<uchar>(i+colunas/2,j+linhas/2);
	image.at<uchar>(i+colunas/2,j+linhas/2) = aux;
    }
  }

  for(int i=colunas/2;i<colunas;i++){
      for(int j=0;j<linhas/2;j++){
       	aux = image.at<uchar>(i,j);
	image.at<uchar>(i,j) = image.at<uchar>(i-colunas/2,j+linhas/2);
	image.at<uchar>(i-colunas/2,j+linhas/2) = aux;
    }
  }

  imshow("Troca Regiões", image);
  imwrite("imagem_trocada.png", image);
  waitKey();

  return 0;
}

2. Preechendo Regiões

2.1. Contando Objetos em uma Cena

O terceiro exercício baseia-se no uso de contagem de objetos em uma cena. Para isso, utilizamos uma imagem fornecida pelo professor, a qual podemos contar 'bolhas'. Existem na imagem dois tipos de bolhas, aquelas que tem buracos e aquelas que são totalmente preenchidas.

Para proceder com a operação, utilizamos da função floodFill() que percorre uma determinada região a partir de um pixel inicial, chamado semente e aquela região de pixels vizinhos recebe a cor determinada no parâmetro da função. Como na imagem as cores eram de valor de pixel 0 ou 255 foi um facilitador para a resolução do problema da contagem de bolhas.

A primeira providência a ser tomada foi excluir as bolhas que estão nas bordas, devido à incerteza de elas serem bolhas preenchidas ou não. Em seguida, aplica-se então a função floodFill() no fundo da imagem para diferenciar as regiões de bolha e buraco da região de fundo.

Feito isto, podemos começar a contar as bolhas que existem com buracos e as que não possuem buraco.

Código

Código 3 - labeling.cpp
#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

int main(int argc, char** argv){
  Mat image;
  int width, height;
  int nobjects, nburacos;
  CvPoint p;
  image = imread("bolhas.png",CV_LOAD_IMAGE_GRAYSCALE);

  if(!image.data){
    std::cout << "imagem nao carregou corretamente\n";
    return(-1);
  }
  width=image.size().width;
  height=image.size().height;

  p.x=0;
  p.y=0;

  // remover objetos das bordas verticais
   for(int i=0; i<height; i++){
	if(image.at<uchar>(i,0) == 255){
		p.x = 0;
		p.y = i;
		floodFill(image,p,0);
	}

	if(image.at<uchar>(i,width-1) == 255){
		p.x = width-1;
		p.y = i;
		floodFill(image,p,0);
	}
   }

  // remover objetos das bordas horizontais
   for(int i=1; i<width-1; i++){
	if(image.at<uchar>(0,i) == 255){
		p.x = i;
		p.y = 0;
		floodFill(image,p,0);
	}

	if(image.at<uchar>(height-1,i) == 255){
		p.x = i;
		p.y = height-1;
		floodFill(image,p,0);
	}
   }

  p.x = 0;
  p.y = 0;
  floodFill(image,p,120);

  // busca objetos
  nobjects=0;
  for(int i=0; i<height; i++){
    for(int j=0; j<width; j++){
      if(image.at<uchar>(i,j) == 255){
		// achou um objeto
		nobjects++;
		p.x=j;
		p.y=i;
		floodFill(image,p,50);
	  }
     }
  }

  // numero de buracos
  nburacos=0;
  for(int i=0; i<height; i++){
    for(int j=0; j<width; j++){
      if(image.at<uchar>(i,j) == 0){
		// achou um objeto
		nburacos++;
		p.x=j;
		p.y=i;
		floodFill(image,p,200);
	  }
     }
  }

  imshow("image", image);
  imwrite("labeling.png", image);
  waitKey();


  cout << "Numero de objetos " << nobjects << endl;
  cout << "Numero de buracos " << nburacos << endl;
  return 0;
}

3. Manipulação de Histogramas

3.1. Equalização

O quarto exercício nos apresenta o conceito de histogramas e como tratá-los. Além disso, utilizamos a câmera de vídeo conectada ao sistema para captura de imagens a partir dos recursos do OpenCV.

Histogramas são dados sobre a imagem organizados em forma de faixas de valores. Como até o momento estamos trabalhando com tons de cinza, o histograma para imagens desse formato consiste em exibir todos os tons de cinza presentes na imagem.

Para uma primeira utilização de histogramas fizemos a equalização das imagens capturadas pela câmera em tempo real. O OpenCV fornece a função equalizeHist() que a partir de uma imagem como fonte e outra como destino, aplica a equalização nos exibindo a imagem original e também equalizada.

A equalização resulta em uma melhor distribuição dos tons de cinza presentes na imagem, como vimos, o histograma documenta a ocorrência dos tons de cinza para imagens em grey scale, sendo assim, esses tons que podem exibir picos no histograma antes da equalização, ficam exibidos de maneira mais distribuída após a equalização, dando um efeito de transição de tons de maneira não brusca.

Código

Código 4 - equalize.cpp
#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

int main(int argc, char** argv){

  Mat image, image_eq, hist, hist_eq;
  VideoCapture cap;
  int nbins = 64;
  float range[] = {0, 256};
  const float *histrange = { range };
  int key;

  bool uniform = true;
  bool acummulate = false;

  cap.open(0);

  if(!cap.isOpened()){
    cout << "cameras indisponiveis";
    return -1;
  }

  int histw = nbins, histh = nbins/2;
  Mat histImg(histh, histw, CV_8UC1, Scalar(0,0,0));
  Mat histImg_eq(histh, histw, CV_8UC1, Scalar(0,0,0));

  while(true){

    cap >> image;
    cvtColor(image, image, CV_BGR2GRAY);
    equalizeHist(image, image_eq);

    calcHist(&image, 1, 0, Mat(), hist, 1,
             &nbins, &histrange,
             uniform, acummulate);
    calcHist(&image_eq, 1, 0, Mat(), hist_eq, 1,
             &nbins, &histrange,
             uniform, acummulate);


    normalize(hist, hist, 0, histImg.rows, NORM_MINMAX, -1, Mat());
    normalize(hist_eq, hist_eq, 0, histImg_eq.rows, NORM_MINMAX, -1,
Mat());

    histImg.setTo(Scalar(0));
    histImg_eq.setTo(Scalar(0));

    for(int i=0; i<nbins; i++){
      line(histImg,
           Point(i, histh),
           Point(i, histh-cvRound(hist.at<float>(i))),
           Scalar(255, 255, 255), 1, 8, 0);
      line(histImg_eq,
           Point(i, histh),
           Point(i, histh-cvRound(hist_eq.at<float>(i))),
           Scalar(255, 255, 255), 1, 8, 0);
    }

    histImg.copyTo(image(Rect(0, 0,nbins, histh)));
    histImg_eq.copyTo(image_eq(Rect(0, 0,nbins, histh)));

    imshow("Imagem Original", image);
    imshow("Imagem Equalizada", image_eq);
    imwrite("imagemOriginal.png", image);
    imwrite("imagemEqualizada.png", image_eq);
    key = waitKey(30);
  }
  return 0;
}

3.2. Comparando histogramas - Motion Detector

Continuando com a manipulação com histogramas agora vamos visualizar uma prática de comparação de histogramas, tendo como finalidade detectar movimento. Para isso, continuamos com a captura de imagens pela câmera em tempo real a fim de obter várias imagens.

O objetivo é que o histograma seja calculado continuamente durante a captura de imagens e utilizando-se apenas de uma das três componentes do histograma já é possível realizar a comparação.

Com essa comparação temos informação de movimento, sendo detectado quando um pixel foi deslocado na imagem, ou seja, a comparação entre os histogramas tem como objetivo obter a diferença do valor de um pixel entre a imagem atual e uma anterior.

Escolhemos ainda uma diferença crítica para que um alerta seja disparado indicando que ocorreu um movimento.

Código

int width, height;

  bool acummulate = false;

  cap.open("video.mp4");

  if(!cap.isOpened()){
    cout << "cameras indisponiveis";
    return -1;
  }

width=cap.get(CV_CAP_PROP_FRAME_WIDTH);
  height=cap.get(CV_CAP_PROP_FRAME_HEIGHT);

#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

void printmask(Mat &m){
  for(int i=0; i<m.size().height; i++){
    for(int j=0; j<m.size().width; j++){
      cout << m.at<float>(i,j) << ",";
    }
    cout << endl;
  }
}

void printmaskLG(Mat &g, Mat &l){
  for(int i=0; i<g.size().height; i++){
    for(int j=0; j<g.size().width; j++){
      cout << g.at<float>(i,j) + l.at<float>(i,j)<< ",";
    }
    cout << endl;
  }
}

void menu(){

	"a - calcular modulo\n"

	"h - horizontal\n"

5.1. Tilt-Shift

As filtragens foram divididas em duas partes, para o início da segunda parte vimos o conceito de tilt-shift.Tilt-shift é uma técnica de fotografia, está presente em lentes para câmeras profissionais e semi-profissionais e envolve rotações entre o plano de foco e o plano de projeção, com o objetivo de desfocar uma região específica da fotografia.

Código

Código - tiltshift.cpp
#include <iostream>
#include <opencv2/opencv.hpp>

using namespace cv;
using namespace std;

double decaimento;
int decaimento_slider = 0;
int decaimento_slider_max = 100;

int center_slider = 0;
int center_slider_max = 100;

int posicao_slider = 0;
int posicao_slider_max = 100;

float media[] = {1,1,1,1,1,1,1,1,1};

Mat image1, image2, blended;
Mat imageTop;

char TrackbarName[50];

void on_trackbar_blend(int, void*){
 decaimento = (double) decaimento_slider/decaimento_slider_max ;
 addWeighted( image1, decaimento, imageTop, 1-decaimento, 0.0, blended);
 imshow("addweighted", blended);
}

void on_trackbar(int, void*){
  image2.copyTo(imageTop);

  Size tamanho = image2.size();
  int largura = tamanho.width;
  int altura = tamanho.height;
  int altura_posicao = (posicao_slider*largura)/100;
  int limite = (center_slider*largura)/100;
  if(altura_posicao > 0){
     if(limite>=0 && limite <= altura-altura_posicao){
        Mat tmp = image1(Rect(0, limite, largura, altura_posicao));
	tmp.copyTo(imageTop(Rect(0, limite, largura,altura_posicao)));
      }else{
        Mat tmp = image1(Rect(0,altura - altura_posicao,largura, altura_posicao));
        tmp.copyTo(imageTop(Rect(0,altura-altura_posicao, largura, altura_posicao)));
     }
 }
  on_trackbar_blend(decaimento_slider,0);
}

int main(int argvc, char** argv){
  image1 = imread("cafe.jpg");
  image2 = image1.clone();

  Mat aux, mask, mask1;


  mask = Mat(3, 3, CV_32F, media);
  scaleAdd(mask, 1/9.0, Mat::zeros(3,3,CV_32F), mask1);
  swap(mask, mask1);
  image2.convertTo(aux, CV_32F);

  for (int i = 0; i < 10; i++)
        filter2D(aux, aux, aux.depth(), mask, Point(1, 1), 0);
    aux=abs(aux);
    aux.convertTo(image2, CV_8UC3);
    image1.copyTo(imageTop);

  namedWindow("addweighted", 1);

  sprintf( TrackbarName, "Decaimento");
  createTrackbar( TrackbarName, "addweighted",
				  &decaimento_slider,
				  decaimento_slider_max,
				  on_trackbar_blend );
  on_trackbar_blend(decaimento_slider, 0 );

  sprintf( TrackbarName, "Posicao Altura" );
  createTrackbar( TrackbarName, "addweighted",
				  &posicao_slider,
				  posicao_slider_max,
				  on_trackbar);
  on_trackbar(posicao_slider, 0 );

  sprintf( TrackbarName, "Centro");
  createTrackbar( TrackbarName, "addweighted",
				  &center_slider,
				  center_slider_max,
				  on_trackbar);

  waitKey(0);
  return 0;
}

	  }

  	width=video.get(CV_CAP_PROP_FRAME_WIDTH);
  	height=video.get(CV_CAP_PROP_FRAME_HEIGHT);
  	Size frameSize(static_cast<int>(width), static_cast<int>(height));
  	VideoWriter out ("saidaVideo.avi", CV_FOURCC('P','I','M','1'), 5, frameSize, true);

	Mat img_in, img_discard;
	int cols = video.get(CV_CAP_PROP_FRAME_WIDTH);
	int rows = video.get(CV_CAP_PROP_FRAME_HEIGHT);
        int key;

	Mat img_blurred = Mat::zeros(rows, cols, CV_32FC3);
	Mat img_out = Mat::zeros(rows, cols, CV_32FC3);
	Mat img_focused = Mat::zeros(rows, cols, CV_32FC3);
	Mat img_unfocused = Mat::zeros(rows, cols, CV_32FC3);
	Mat mask_focus = Mat(rows, cols, CV_32FC3, Scalar(0, 0, 0));
	Mat mask_unfocus = Mat(rows, cols, CV_32FC3, Scalar(1, 1, 1));

	int frame_skip = 2;
	int pass_frame = 0;

	float unfocus_aux = 0.35;
	int pos_i = unfocus_aux*rows;
	int pos_f = rows - unfocus_aux*rows;
	int decay = 70;

	Vec3f mask_function_value;

	for (int i = 0; i < rows; i++) {
		for (int j = 0; j < cols; j++) {
			mask_function_value[0] = (tanh((float(i - pos_i) / decay)) - tanh((float(i - pos_f) / decay))) / 2;
			mask_function_value[1] = (tanh((float(i - pos_i) / decay)) - tanh((float(i - pos_f) / decay))) / 2;
			mask_function_value[2] = (tanh((float(i - pos_i) / decay)) - tanh((float(i - pos_f) / decay))) / 2;
			mask_focus.at<Vec3f>(i, j) = mask_function_value;
		}
	}
	mask_unfocus = mask_unfocus - mask_focus;


	while (true) {
		if (pass_frame == 0) {
			video >> img_in;
		} else {
			video >> img_discard;
		}

		img_in.convertTo(img_in, CV_32FC3);
		GaussianBlur(img_in, img_blurred, Size(7, 7), 0, 0);

		img_focused = img_in.mul(mask_focus);
		img_unfocused = img_blurred.mul(mask_unfocus);

		img_blurred.convertTo(img_blurred, CV_8UC3);

		img_out = img_focused + img_unfocused;
		img_out.convertTo(img_out, CV_8UC3);

		img_focused.convertTo(img_focused, CV_8UC3);
		img_unfocused.convertTo(img_unfocused, CV_8UC3);
		imshow("img_focused", img_focused);
		imshow("img_unfocused", img_unfocused);

		if (pass_frame == 0) {
			pass_frame = frame_skip;
		}
		else {
			pass_frame -= 1;
		}

		imshow("Efeito tiltshift", img_out);
		key = waitKey(30);
	}
	return 0;
}
